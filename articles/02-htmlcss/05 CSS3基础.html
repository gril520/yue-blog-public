html
<!DOCTYPE html>
<html lang="zh-CN">
<head>
    <meta charset="UTF-8">
    <title>01_线性渐变</title>
    <style>
        .box {
            width: 300px;
            height: 200px;
            border: 1px solid black;
            float: left;
            margin-left: 50px;
            font-size: 20px;
        }
        .box1 {
            /* 线性渐变 默认从上到下*/
            background-image: linear-gradient(red,yellow,green);

        }
        .box4 {
            background-image: linear-gradient(red 50px,yellow 100px,green 150px);

            text-align: center;
            line-height: 200px;
            font-weight: bold;
            color: transparent;
            /* 让背景只呈现在文字上 */
            -webkit-background-clip: text;
        }
    </style>
</head>

             /* 不一定为正圆，与容器形状有关 */
            background-image: radial-gradient(red,yellow,green);

        }
        .box3 {
